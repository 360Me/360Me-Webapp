var FeedbackActions = _interopRequire(require("../actions/FeedbackActions"));
 
var FeedbackForm = React.createClass({
  displayName: "FeedbackForm",
 
  propTypes: {
    to: React.PropTypes.string.isRequired,
    closeModal: React.PropTypes.func
  },
 
  getDefaultProps: function getDefaultProps() {
    people: [];
  },
 
  componentDidMount: function componentDidMount() {},

    var feedback = {
      to: this.props.to,
      text: this.refs.text.getDOMNode().value
    };
    FeedbackActions.createFeedback(feedback);
 
    if (this.props.closeModal) this.props.closeModal();
  },

    return React.createElement(
      "div",
      null,
      React.createElement("textarea", { ref: "text", autoFocus: true, placeholder: "Enter some feedback" }),
      React.createElement(
        "button",
        { onClick: this.handleCreateFeedback },
        "Send feedback"
      )
    );
  }
});
